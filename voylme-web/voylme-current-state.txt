function valueToText(value: unknown): string {
  if (typeof value === "string") {
    const trimmed = value.trim();

    if (
      trimmed.startsWith("{") &&
      trimmed.endsWith("}")
    ) {
      try {
        return valueToText(JSON.parse(trimmed));
      } catch {
        return trimmed;
      }
    }

    return trimmed;
  }

  if (
    value &&
    typeof value === "object"
  ) {
    const item = value as Record<string, unknown>;

    return [
      item.code,
      item.languageCode,
      item.locale,
      item.value,
      item.name,
      item.label,
      item.language,
    ]
      .filter(Boolean)
      .map(String)
      .join(" ");
  }

  return "";
}

function resolveHeroLocale(
  value: unknown,
): HeroLocale {
  const code = valueToText(value)
    .trim()
    .toLowerCase();

  if (
    code === "ar" ||
    code === "ara" ||
    code.startsWith("ar-") ||
    code.includes("arabic") ||
    code.includes("العربية") ||
    code.includes("عربي")
  ) {
    return "ar";
  }

  if (
    code === "fr" ||
    code === "fra" ||
    code === "fre" ||
    code.startsWith("fr-") ||
    code.includes("french") ||
    code.includes("français") ||
    code.includes("francais")
  ) {
    return "fr";
  }

  return "en";
}

export default function Hero() {
  const [locale, setLocale] =
    useState<HeroLocale>("en");

  useEffect(() => {
    const syncLanguage = () => {
      const storedLanguage =
        localStorage.getItem(
          "voylme-language",
        );

      setLocale(
        resolveHeroLocale(storedLanguage),
      );
    };

    const handleLanguageChange = (
      event: Event,
    ) => {
      const customEvent =
        event as CustomEvent<unknown>;

      setLocale(
        resolveHeroLocale(
          customEvent.detail,
        ),
      );
    };

    syncLanguage();

    const languageSyncInterval = window.setInterval(syncLanguage, 250);

    window.addEventListener(
      "voylme-language-change",
      handleLanguageChange,
    );

    window.addEventListener(
      "storage",
      syncLanguage,
    );

    window.addEventListener(
      "focus",
      syncLanguage,
    );

    return () => {
      window.clearInterval(languageSyncInterval);

      window.removeEventListener(
        "voylme-language-change",
        handleLanguageChange,
      );

      window.removeEventListener(
        "storage",
        syncLanguage,
      );

      window.removeEventListener(
        "focus",
        syncLanguage,
      );
    };
  }, []);

  const content = HERO_TEXT[locale];

  return (
    <section className="home-hero">
      <div className="home-hero-frame voylme-final-overlay-hero">
        <img
          src="/images/hero-bg-clean.png?v=hero-final-20260713"
          alt=""
          aria-hidden="true"
          draggable={false}
        />

        <div
          key={locale}
          className="voylme-final-overlay-hero-copy"
          dir={content.dir}
          lang={locale}
        >
          <p className="voylme-final-overlay-hero-eyebrow">
            {content.eyebrow}
          </p>

          <h2>{content.title}</h2>

          <p className="voylme-final-overlay-hero-description">
            {content.description}
          </p>
        </div>
      </div>
    </section>
  );
}
